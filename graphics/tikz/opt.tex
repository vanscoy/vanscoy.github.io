  \begin{axis}[xmin=-3,xmax=2,line width=1.3mm,axis lines=none]

    \addplot [domain=1:2] {x*(x-1)-1};
    \draw (-2,0) -- (1,-1) node[point,at start] {} node[point,at end] {};
